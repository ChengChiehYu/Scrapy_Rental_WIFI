def get_current_ip():
    res = requests.get('http://ipinfo.io')
    soup = BeautifulSoup(res.text,'lxml')
    ip = soup.select('p')[0]
    res = re.search("([0-9]{1,3}\.){3}[0-9]{1,3}", ip.text)
    return res.group()
